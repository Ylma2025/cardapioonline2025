    <script>
        // URL do arquivo CSV no GitHub
        const CSV_URL = 'https://raw.githubusercontent.com/ylma2025/cardapioonline2025/main/cardapio.csv';
        
        // Cache para melhor performance
        let cache = {
            data: null,
            timestamp: null,
            timeout: 5 * 60 * 1000 // 5 minutos
        };

        function formatarPreco(preco) {
            if (preco === 'variable' || preco === '') return 'Sob Consulta';
            const numero = parseFloat(preco);
            return isNaN(numero) ? preco : `R$ ${numero.toFixed(2).replace('.', ',')}`;
        }

        function limparDescricao(descricao) {
            if (!descricao) return '';
            // Remove tags HTML e espa√ßos extras
            return descricao.replace(/<[^>]*>/g, '').replace(/\s+/g, ' ').trim();
        }

        async function carregarCardapio() {
            const container = document.getElementById('cardapio-container');
            
            try {
                // Verificar cache
                if (cache.data && cache.timestamp && (Date.now() - cache.timestamp) < cache.timeout) {
                    console.log('Usando dados em cache');
                    renderizarCardapio(cache.data);
                    return;
                }

                container.innerHTML = '<div class="loading">üîÑ Atualizando card√°pio...</div>';

                // Buscar CSV do GitHub
                const response = await fetch(CSV_URL);
                
                if (!response.ok) {
                    throw new Error(`Erro ao carregar: ${response.status}`);
                }
                
                const csvText = await response.text();
                const cardapioData = processarCSV(csvText);
                
                // Atualizar cache
                cache.data = cardapioData;
                cache.timestamp = Date.now();
                
                renderizarCardapio(cardapioData);
                
            } catch (error) {
                console.error('Erro ao carregar card√°pio:', error);
                container.innerHTML = `
                    <div class="error">
                        ‚ùå Erro ao carregar card√°pio: ${error.message}
                        <br><br>
                        <button onclick="carregarCardapio()" style="
                            padding: 10px 20px;
                            background: #8b5fad;
                            color: white;
                            border: none;
                            border-radius: 5px;
                            cursor: pointer;
                        ">üîÑ Tentar Novamente</button>
                    </div>
                `;
            }
        }

        function processarCSV(csvText) {
            const linhas = csvText.split('\n').filter(linha => linha.trim() !== '');
            const headers = linhas[0].split(',').map(h => h.trim());
            
            const cardapioData = {};
            
            for (let i = 1; i < linhas.length; i++) {
                const valores = parseCSVLinha(linhas[i]);
                const item = {};
                
                for (let j = 0; j < headers.length; j++) {
                    item[headers[j]] = valores[j] ? valores[j].trim() : '';
                }
                
                // Processar apenas itens dispon√≠veis para venda
                if (item['Available for sale [Ylma Cake\'s]'] === 'Y') {
                    const categoria = item.Category || 'Geral';
                    const estoque = parseFloat(item['In stock [Ylma Cake\'s]']) || 0;
                    
                    const itemFormatado = {
                        nome: item.Name || 'Item sem nome',
                        descricao: limparDescricao(item.Description),
                        preco: item['Price [Ylma Cake\'s]'] || '0',
                        estoque: estoque,
                        semEstoque: estoque <= 0
                    };
                    
                    // Adicionar √† categoria correspondente
                    if (!cardapioData[categoria]) {
                        cardapioData[categoria] = [];
                    }
                    cardapioData[categoria].push(itemFormatado);
                }
            }
            
            return cardapioData;
        }

        function parseCSVLinha(linha) {
            const valores = [];
            let valorAtual = '';
            let dentroDeAspas = false;
            
            for (let i = 0; i < linha.length; i++) {
                const char = linha[i];
                
                if (char === '"') {
                    dentroDeAspas = !dentroDeAspas;
                } else if (char === ',' && !dentroDeAspas) {
                    valores.push(valorAtual);
                    valorAtual = '';
                } else {
                    valorAtual += char;
                }
            }
            
            valores.push(valorAtual);
            return valores;
        }

        function renderizarCardapio(cardapioData, categoriaFiltro = 'all') {
            const container = document.getElementById('cardapio-container');
            container.innerHTML = '';

            let temItens = false;

            Object.entries(cardapioData).forEach(([categoria, itens]) => {
                if (categoriaFiltro !== 'all' && categoriaFiltro !== categoria) return;

                // S√≥ mostra categoria se tiver itens
                if (itens.length > 0) {
                    temItens = true;
                    
                    const categoriaSection = document.createElement('div');
                    categoriaSection.className = 'categoria-section';
                    
                    const titulo = document.createElement('h2');
                    titulo.className = 'categoria-titulo';
                    titulo.textContent = categoria;
                    categoriaSection.appendChild(titulo);

                    const grid = document.createElement('div');
                    grid.className = 'cardapio-grid';

                    itens.forEach(item => {
                        const itemCard = document.createElement('div');
                        itemCard.className = `item-card ${item.semEstoque ? 'sem-estoque' : ''}`;
                        
                        itemCard.innerHTML = `
                            <div class="item-header">
                                <h3 class="item-nome">${item.nome}</h3>
                                <span class="item-preco">${formatarPreco(item.preco)}</span>
                            </div>
                            ${item.descricao ? `<p class="item-descricao">${item.descricao}</p>` : ''}
                            ${item.estoque > 0 ? `<small style="color: #27ae60;">‚úÖ Em estoque</small>` : ''}
                        `;

                        grid.appendChild(itemCard);
                    });

                    categoriaSection.appendChild(grid);
                    container.appendChild(categoriaSection);
                }
            });

            if (!temItens) {
                container.innerHTML = '<div class="error">üìù Nenhum item dispon√≠vel no momento</div>';
            }
        }

        // Filtros
        function aplicarFiltro(categoria) {
            document.querySelectorAll('.filtro-btn').forEach(b => b.classList.remove('active'));
            event.target.classList.add('active');
            renderizarCardapio(cache.data, categoria);
        }

        document.addEventListener('DOMContentLoaded', function() {
            // Configurar eventos dos filtros
            document.querySelectorAll('.filtro-btn').forEach(btn => {
                btn.addEventListener('click', function() {
                    aplicarFiltro(this.getAttribute('data-categoria'));
                });
            });

            // Carregar card√°pio inicial
            carregarCardapio();

            // Atualizar automaticamente a cada 5 minutos
            setInterval(carregarCardapio, 5 * 60 * 1000);
        });

        // Recarregar quando a p√°gina ganha foco (usu√°rio volta para a aba)
        document.addEventListener('visibilitychange', function() {
            if (!document.hidden) {
                carregarCardapio();
            }
        });
    </script>
